Automated Response Example

def lambda_handler(event, context):
    # Parse GuardDuty finding
    finding = event['detail']['findings'][0]
    threat_level = finding['Severity']
    
    if threat_level >= 7:
        # High severity threat
        block_ip(finding['Source']['IpAddress'])
        notify_security_team(finding)
        create_security_ticket(finding)
    
    # Log incident
    log_security_event(finding)
    
    return {
        'statusCode': 200,
        'body': 'Incident handled successfully'
    }
